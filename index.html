function capturarYEnviar() { const mensaje = document.getElementById('mensaje'); const loader = document.getElementById('loader'); const urlParams = new URLSearchParams(window.location.search); const nombreEmpleado = urlParams.get('id') || "Empleado_Sin_Nombre"; const linkMapsDiv = document.getElementById('link-maps'); linkMapsDiv.innerHTML = ""; if (!navigator.geolocation) { mensaje.innerText = "Error: Su teléfono no permite GPS desde el navegador."; mensaje.style.color = "red"; return; } // Detectar idioma del navegador const lang = navigator.language || navigator.userLanguage; // Mensaje localizado let locatingMsg = "Localizando dispositivo..."; if (lang.startsWith("zh")) { locatingMsg = "正在定位设备..."; } mensaje.innerText = locatingMsg; mensaje.style.color = "#555"; loader.style.display = "block"; navigator.geolocation.getCurrentPosition((posicion) => { const lat = posicion.coords.latitude; const lon = posicion.coords.longitude; const scriptURL = "https://script.google.com/macros/s/AKfycbzjc5-YSH9bfF-bcKvsFn6TxoC4v7_qk2dQ3qBFfxpd4nF12PLKZz4bx5GjEVv11BWN/exec"; fetch(`${scriptURL}?nombre=${encodeURIComponent(nombreEmpleado)}&lat=${lat}&lon=${lon}`, { mode: 'no-cors' }) .then(() => { loader.style.display = "none"; mensaje.innerText = "✔ Ubicación enviada con éxito. Puede cerrar esta ventana."; mensaje.style.color = "green"; const link = document.createElement("a"); link.href = `https://maps.google.com/?q=${lat},${lon}`; link.target = "_blank"; link.rel = "noopener"; link.innerText = "Ver mi ubicación en Google Maps"; link.style.display = 'inline-block'; link.style.marginTop = '18px'; link.style.fontSize = '16px'; link.style.color = '#1565c0'; link.style.textDecoration = 'underline'; linkMapsDiv.appendChild(link); }) .catch(error => { loader.style.display = "none"; mensaje.innerText = "Error en la red. Intente de nuevo."; mensaje.style.color = "red"; }); }, (error) => { loader.style.display = "none"; mensaje.innerText = "Error: Debe aceptar el permiso de GPS para completar el registro."; mensaje.style.color = "red"; }, { enableHighAccuracy: true, timeout: 10000 }); }
